675  flv_stream_builder &init_stream_header(std::vector<uint8_t> &buf,
676  bool has_audio, bool has_video) {
677  buf.clear();
678  buf.resize(9 + 4, 0);
679 
680  uint8_t flags = 0;
681  has_audio_ = has_audio;
682  if (has_audio_) {
683  flags |= 0x04;
684  }
685  has_video_ = has_video;
686  if (has_video_) {
687  flags |= 0x01;
688  }
689 
690  // Signature
691  buf[0] = 'F';
692  buf[1] = 'L';
693  buf[2] = 'V';
694 
695  // Version
696  buf[3] = 0x01;
697 
698  // Flags
699  buf[4] = flags;
700 
701  // Header size
702  buf[5] = 0;
703  buf[6] = 0;
704  buf[7] = 0;
705  buf[8] = FLV_HEADER_SIZE;
706 
707  buf[9] = 0;
708  buf[10] = 0;
709  buf[11] = 0;
710  buf[12] = 0;
711  return *this;
712  }

900  void append_tag(std::vector<uint8_t> &buf, tag_type_t type,
901  uint32_t timestamp, uint32_t strem_id, const uint8_t *data,
902  uint32_t length) {
903  int32_t original_size = buf.size();
904  buf.reserve(buf.size() + FLV_TAG_HEADER_SIZE + length);
905 
906  // Header.Type
907  buf.emplace_back(type);
908 
909  // Header.DataSize
910  buf.emplace_back((length & 0x00ff0000) >> 16);
911  buf.emplace_back((length & 0x0000ff00) >> 8);
912  buf.emplace_back((length & 0x000000ff));
913 
914  // Header.Timestamp
915  buf.emplace_back((timestamp & 0x00ff0000) >> 16);
916  buf.emplace_back((timestamp & 0x0000ff00) >> 8);
917  buf.emplace_back((timestamp & 0x000000ff));
918 
919  // Header.TimestampExtended
920  buf.emplace_back((timestamp & 0xff000000) >> 24);
921 
922  // Header.StreamID (actually this is always 0 according to the
923  // specification)
924  buf.emplace_back((strem_id & 0x00ff0000) >> 16);
925  buf.emplace_back((strem_id & 0x0000ff00) >> 8);
926  buf.emplace_back((strem_id & 0x000000ff));
927 
928  // Data
929  std::copy(data, data + length, std::back_inserter(buf));
930 
931  uint32_t tag_size = buf.size() - original_size;
932  append_tag_size(buf, tag_size);
933 
934  tag_count_++;
935  }
